### 2.2 ⚠️ Needs Clarification — Integration Points

#### Issue 1: InitiativeEngine Persistence

**Current:** InitiativeEngine has in-memory `_initiatives` list

**Spec says:** "Modify InitiativeEngine to use InitiativeStore"

**Problem:** The spec doesn't clarify the relationship between:
- `InitiativeEngine._initiatives` (in-memory)
- `InitiativeStore` (SQLite)
- The initiatives generated by `engine.generate()`

**Resolution:**
```python
class InitiativeEngine:
    def __init__(self, ..., store: Optional["InitiativeStore"] = None):
        self._store = store
        self._initiatives: List[Initiative] = []  # Keep for backward compat
    
    async def generate(self, ...) -> List[Initiative]:
        initiatives = await self._generate_all(types)
        
        # Persist to SQLite if store available
        if self._store:
            for init in initiatives:
                # Check dedup_key before creating
                existing = await self._store.get_by_dedup_key(init.dedup_key)
                if not existing:
                    await self._store.create(init)
        
        # Still return for immediate use
        self._initiatives.extend(initiatives)
        return sorted(initiatives, ...)
    
    async def get_active(self) -> List[Initiative]:
        # Prefer SQLite if available
        if self._store:
            return await self._store.list(status=["pending", "assigned"])
        # Fallback to in-memory for backward compat
        return self._initiatives
```

#### Issue 3: Delivery Bridge Agent Routing

**Current:** `push_initiative()` pushes to OpenClaw gateway only

**Spec says:** Route to agents based on assignment

**Problem:** The bridge doesn't know about agents

**Resolution:**
```python
class ProactiveDeliveryBridge:
    def __init__(
        self,
        ...,
        agent_store: Optional["AgentStore"] = None,
        websocket_manager: Optional["WebSocketManager"] = None,
    ):
        self._agent_store = agent_store
        self._ws_manager = websocket_manager
    
    async def push_initiative(self, initiative: Dict) -> bool:
        """Push initiative to assigned agent or gateway."""
        
        assigned_agent_id = initiative.get("assigned_agent_id")
        
        # If assigned to specific agent, route to them
        if assigned_agent_id and self._agent_store:
            agent = await self._agent_store.get(assigned_agent_id)
            if agent:
                return await self._deliver_to_agent(initiative, agent)
        
        # Fallback: push to gateway (OpenClaw main session)
        return await self._push_to_gateway(initiative)
    
    async def _deliver_to_agent(self, initiative: Dict, agent: Agent) -> bool:
        if agent.connection_mode == "local" and agent.gateway_url:
            return await self._http_push(agent.gateway_url, initiative)
        elif agent.websocket_connected and self._ws_manager:
            return await self._ws_push(agent.agent_id, initiative)
        return False
```

```python
# agents/websocket.py
from fastapi import WebSocket
from typing import Dict, Set
import asyncio

class WebSocketManager:
    def __init__(self):
        self._connections: Dict[str, WebSocket] = {}
        self._lock = asyncio.Lock()
    
    async def connect(self, agent_id: str, websocket: WebSocket):
        await websocket.accept()
        async with self._lock:
            self._connections[agent_id] = websocket
    
    async def disconnect(self, agent_id: str):
        async with self._lock:
            self._connections.pop(agent_id, None)
    
    async def send(self, agent_id: str, message: dict) -> bool:
        async with self._lock:
            ws = self._connections.get(agent_id)
            if ws:
                try:
                    await ws.send_json(message)
                    return True
                except:
                    self._connections.pop(agent_id, None)
        return False
    
    async def broadcast(self, message: dict) -> int:
        """Broadcast to all connected agents. Returns count reached."""
        count = 0
        async with self._lock:
            for agent_id, ws in list(self._connections.items()):
                try:
                    await ws.send_json(message)
                    count += 1
                except:
                    self._connections.pop(agent_id, None)
        return count

# Global instance
_ws_manager: Optional[WebSocketManager] = None

def set_ws_manager(manager: WebSocketManager) -> None:
    global _ws_manager
    _ws_manager = manager
```

async def agent_websocket(websocket: WebSocket, agent_id: str):
    global _ws_manager, _agent_store
    
    # Auth via header
    auth = websocket.headers.get("Authorization", "")
    if not auth.startswith("Bearer "):
        await websocket.close(code=4001, reason="Unauthorized")
        return
    
    # Validate agent exists
    if not _agent_store:
        await websocket.close(code=5001, reason="Agent store not available")
        return
    
    agent = await _agent_store.get(agent_id)
    if not agent:
        await websocket.close(code=4004, reason="Agent not found")
        return
    
    # Connect
    await _ws_manager.connect(agent_id, websocket)
    await _agent_store.update(agent_id, websocket_connected=True, status="online")
    
    try:
        # Start heartbeat task
        heartbeat_task = asyncio.create_task(_send_heartbeat(websocket))
        
        # Message loop
        while True:
            data = await websocket.receive_json()
            await _handle_agent_message(agent_id, data)
    except WebSocketDisconnect:
        pass
    finally:
        heartbeat_task.cancel()
        await _ws_manager.disconnect(agent_id)
        await _agent_store.update(agent_id, websocket_connected=False, status="offline")

### 3.2 New Phases Implementation

```python
async def _phase_initiative_assignment(self) -> None:
    """Assign pending initiatives to available agents."""
    store = self._registry.initiative_store
    agent_store = self._registry.agent_store
    
    if not store or not agent_store:
        return
    
    try:
        # Get pending initiatives
        pending = await store.list(status="pending", limit=50)
        
        # Get online agents
        agents = await agent_store.list(status="online")
        
        for initiative in pending:
            # Check if already assigned
            if initiative.get("assigned_agent_id"):
                continue
            
            # Select best agent
            agent = select_agent_for_initiative(initiative, agents)
            
            if agent:
                # Atomic assignment
                updated = await store.assign(initiative["id"], agent["agent_id"])
                if updated:
                    logger.info(
                        "Assigned initiative %s to agent %s",
                        initiative["id"], agent["name"]
                    )
    except Exception as exc:
        self.stats.errors += 1
        logger.error("Phase initiative_assignment error: %s", exc)

async def _phase_agent_heartbeat(self) -> None:
    """Mark agents offline if no heartbeat for 90s."""
    store = self._registry.agent_store
    if not store:
        return
    
    try:
        threshold = datetime.now(timezone.utc) - timedelta(seconds=90)
        stale = await store.list(status="online", last_seen_before=threshold)
        
        for agent in stale:
            await store.update(agent["agent_id"], status="offline")
            logger.info("Agent %s marked offline (no heartbeat)", agent["name"])
            
            # Reassign their assigned initiatives
            await self._reassign_agent_initiatives(agent["agent_id"])
    except Exception as exc:
        self.stats.errors += 1
        logger.error("Phase agent_heartbeat error: %s", exc)

async def _phase_initiative_timeout(self) -> None:
    """Handle timed-out initiatives."""
    store = self._registry.initiative_store
    if not store:
        return
    
    try:
        timeout_threshold = datetime.now(timezone.utc) - timedelta(seconds=300)
        timed_out = await store.list(
            status="assigned",
            assigned_before=timeout_threshold
        )
        
        for initiative in timed_out:
            attempt = initiative.get("attempt_count", 0) + 1
            max_attempts = initiative.get("max_attempts", 3)
            
            if attempt >= max_attempts:
                await store.update(
                    initiative["id"],
                    status="failed",
                    failed_at=datetime.now(timezone.utc),
                    failed_reason="Max attempts reached"
                )
                logger.warning(
                    "Initiative %s failed after %d attempts",
                    initiative["id"], attempt
                )
            else:
                # Reassign to different agent
                await store.update(
                    initiative["id"],
                    status="pending",
                    assigned_agent_id=None,
                    attempt_count=attempt
                )
                logger.info(
                    "Initiative %s timed out, reassigning (attempt %d/%d)",
                    initiative["id"], attempt, max_attempts
                )
    except Exception as exc:
        self.stats.errors += 1
        logger.error("Phase initiative_timeout error: %s", exc)

async def _reassign_agent_initiatives(self, agent_id: str) -> int:
    """Reassign all initiatives from an agent. Returns count reassigned."""
    store = self._registry.initiative_store
    if not store:
        return 0
    
    initiatives = await store.list(assigned_to=agent_id, status="assigned")
    reassigned = 0
    
    for init in initiatives:
        # Increment attempt count
        attempt = init.get("attempt_count", 0) + 1
        
        if attempt >= init.get("max_attempts", 3):
            await store.update(init["id"], status="failed", failed_reason="Agent went offline")
        else:
            await store.update(
                init["id"],
                status="pending",
                assigned_agent_id=None,
                attempt_count=attempt
            )
            reassigned += 1
    
    return reassigned
```

async function connectRemoteAgent(
  config: ColonyPluginConfig,
  api: OpenClawPluginApi,
): Promise<void> {
  // Load agent config from ~/.colony/agent.json
  const agentConfig = await loadAgentConfig();
  
  if (!agentConfig) {
    api.logger?.error?.(
      "[colony] No agent config found. Run 'colony agent connect' first."
    );
    return;
  }
  
  // Connect WebSocket
  const wsUrl = agentConfig.websocket_url;
  const ws = new WebSocket(wsUrl, {
    headers: {
      Authorization: `Bearer ${signNodeCert(agentConfig.node_cert)}`,
      "X-Agent-Id": agentConfig.agent_id,
    },
  });
  
  ws.on("open", () => {
    api.logger?.info?.("[colony] WebSocket connected to Colony");
    startWebSocketHeartbeat(ws, api);
  });
  
  ws.on("message", (data) => {
    const msg = JSON.parse(data.toString());
    handleWebSocketMessage(msg, ws, api);
  });
  
  ws.on("close", () => {
    api.logger?.warn?.("[colony] WebSocket disconnected, reconnecting in 5s...");
    setTimeout(() => connectRemoteAgent(config, api), 5000);
  });
  
  ws.on("error", (err) => {
    api.logger?.error?.(`[colony] WebSocket error: ${err}`);
  });
  
  // Store reference for cleanup
  (api as any)._colonyWebSocket = ws;
}

function startWebSocketHeartbeat(ws: WebSocket, api: OpenClawPluginApi): void {
  const interval = setInterval(() => {
    if (ws.readyState === WebSocket.OPEN) {
      ws.send(JSON.stringify({
        type: "status",
        status: "online",
        current_assignments: (api as any)._colonyAssignments?.size ?? 0,
      }));
    } else {
      clearInterval(interval);
    }
  }, 30000);
}

async function handleWebSocketMessage(
  msg: any,
  ws: WebSocket,
  api: OpenClawPluginApi,
): Promise<void> {
  switch (msg.type) {
    case "initiative": {
      const initiative = msg.initiative;
      
      // Store for tracking
      const assignments = (api as any)._colonyAssignments ?? new Map();
      assignments.set(initiative.id, initiative);
      (api as any)._colonyAssignments = assignments;
      
      // Enqueue as system event for LLM to handle
      const runtime = (api as any).runtime;
      if (runtime?.system?.enqueueSystemEvent) {
        await runtime.system.enqueueSystemEvent({
          type: "colony_initiative",
          initiative,
        });
      }
      
      // Acknowledge
      ws.send(JSON.stringify({
        type: "acknowledge",
        initiative_id: initiative.id,
      }));
      break;
    }
    
    case "ping":
      ws.send(JSON.stringify({
        type: "pong",
        timestamp: new Date().toISOString(),
      }));
      break;
    
    case "config":
      // Update local config from server
      api.logger?.info?.(`[colony] Config update: ${JSON.stringify(msg.config)}`);
      break;
    
    case "disconnect":
      api.logger?.warn?.(`[colony] Server requested disconnect: ${msg.reason}`);
      ws.close();
      break;
  }
}

### 5.1 Setup Code Generation

```python
import secrets
import string

def generate_setup_code() -> str:
    """Generate secure random setup code."""
    chars = string.ascii_uppercase + string.digits
    # Remove ambiguous chars: 0/O, 1/I/L
    chars = chars.replace("O", "").replace("0", "").replace("1", "").replace("I", "").replace("L", "")
    
    segments = [
        "".join(secrets.choice(chars) for _ in range(4))
        for _ in range(3)
    ]
    return "COLONY-" + "-".join(segments)

        return False
    
    # Get agent
    agent = await agent_store.get(agent_id)
    if not agent:
        return False
    
    # Get node public key from cert
    cert = json.loads(agent["node_cert"])
    pubkey = cert["node_public_key_ed25519"]
    
    # Verify signature
    message = f"{agent_id}:{timestamp}"
    return _verify_ed25519_signature(pubkey, message.encode(), signature)
```
